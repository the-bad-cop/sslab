#include < stdio.h> #include < stdlib.h> #include < string.h> #define MAX 5 struct new { int hashkey; int address; char label[10]; } hash[MAX]; void main() { void display(); int check(int); int check_meet(int); int i, key, cnt = 0, add, ch; char lab[10]; for (i = 0; i < MAX; i++) { hash[i].address = -1; hash[i].hashkey = i; } while (1) { printf("Enter the choice: 1:Create 2:Search 3: Exit\n"); scanf("%d", &ch); switch (ch) { case 1: if (cnt == MAX) { printf("\nTable full\n"); display(); } else { printf("Enter the address and label::"); scanf("%d %s", &add, lab); key = add % MAX; if (check(key) == 1) { hash[key].address = add; strcpy(hash[key].label, lab); cnt++; display(); } else { if (hash[key].address == add) { printf("Duplicate address!\n"); break; } else { key = check_meet(key); hash[key].address = add; strcpy(hash[key].label, lab); cnt++; display(); } } } break; case 2: printf("Enter the label:: "); scanf("%s", lab); for (i = 0; i < MAX; i++) { if (strcmp(hash[i].label, lab) == 0) { printf("\nLabel found at %d:%s\n", hash[i].address, hash[i].label); break; } } if (i == MAX) { printf("Label not found!\n"); } break; case 3: exit(0); break; default: break; } } } void display() { int i; printf("HASHKEY\tLABEL\tADDRESS\n"); for (i = 0; i < MAX; i++) { printf("%d\t%s\t%d\n", hash[i].hashkey, hash[i].label, hash[i].address); } } int check(int a) { if (hash[a].address == -1) { return 1; } else { return 0; } } int check_meet(int a) { int i; for (i = a; i < MAX; i++) { if (hash[i].address == -1) { return i; } } for (i = 0; i < a - 1; i++) { if (hash[i].address == -1) { return i; } } }
